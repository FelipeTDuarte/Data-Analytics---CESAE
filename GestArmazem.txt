
CREATE TABLE Fornecedor
(
F_ID INT NOT NULL AUTO_INCREMENT,
F_Nome VARCHAR (60) NOT NULL,
F_Cidade VARCHAR (60) NOT NULL,
PRIMARY KEY (F_ID) );#

CREATE TABLE Produto
(
P_ID INT NOT NULL AUTO_INCREMENT,
P_F_ID INT NOT NULL 
P_Descricao VARCHAR (60) NOT NULL,
P_StockAtual INT NOT NULL,
PRIMARY KEY (P_ID) );#

CREATE TABLE Compra
(
C_ID INT NOT NULL AUTO_INCREMENT,
C_Data DATE NOT NULL
C_F_ID INT NOT NULL
C_P_ID INT NOT NULL
Quantidade INT NOT NULL,
PRIMARY KEY (C_ID) );#
